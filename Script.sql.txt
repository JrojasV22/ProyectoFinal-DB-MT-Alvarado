-- ELIMINACIÓN CONTROLADA DE OBJETOS
-- Permite volver a ejecutar el script durante el desarrollo.
-- =====================================================================

BEGIN
  FOR obj IN (
    SELECT object_name, object_type
    FROM user_objects
    WHERE object_type IN ('TRIGGER', 'TABLE')
      AND object_name IN (
        'TRG_USUARIO_FEC_ACT','TRG_PRODUCTO_FEC_ACT','TRG_SERVICIO_FEC_ACT',
        'TRG_MAQUINARIA_FEC_ACT','TRG_SOLICITUD_FEC_ACT','TRG_CONSULTA_FEC_ACT',
        'TRG_COTIZACION_CORREO','TRG_RESPUESTA_CORREO',
        'RESPUESTA_CONSULTA','CONSULTA_TECNICA','ESTADO_CONSULTA','COTIZACION',
        'SOLICITUD_COTIZACION','ESTADO_SOLICITUD','HISTORIAL_ESTADO_MAQUINARIA',
        'MAQUINARIA','ESTADO_MAQUINARIA','SERVICIO','CATEGORIA_SERVICIO',
        'PRODUCTO','CATEGORIA_PRODUCTO','CLIENTE','ROL_PERMISO','USUARIO_ROL',
        'PERMISO','ROL','USUARIO'
      )
    ORDER BY CASE object_type WHEN 'TRIGGER' THEN 1 WHEN 'TABLE' THEN 2 ELSE 3 END
  ) LOOP
    BEGIN
      IF obj.object_type = 'TRIGGER' THEN
        EXECUTE IMMEDIATE 'DROP TRIGGER ' || obj.object_name;
      ELSIF obj.object_type = 'TABLE' THEN
        EXECUTE IMMEDIATE 'DROP TABLE ' || obj.object_name || ' CASCADE CONSTRAINTS PURGE';
      END IF;
    EXCEPTION
      WHEN OTHERS THEN NULL;
    END;
  END LOOP;
END;
/

CREATE OR REPLACE PROCEDURE PR_MT_REG_USUARIO(p_usuario VARCHAR2,p_correo VARCHAR2,p_hash VARCHAR2,p_nombre VARCHAR2,p_apellidos VARCHAR2,p_telefono VARCHAR2,p_id OUT NUMBER)
IS

BEGIN
INSERT INTO usuario(nombre_usuario,correo,hash_contrasena,nombre,apellidos,telefono)
VALUES(TRIM(p_usuario),LOWER(TRIM(p_correo)),p_hash,TRIM(p_nombre),TRIM(p_apellidos),TRIM(p_telefono))
RETURNING id_usuario INTO p_id;
END;
/

CREATE OR REPLACE PROCEDURE PR_MT_ACT_USUARIO(p_id NUMBER,p_correo VARCHAR2,p_nombre VARCHAR2,p_apellidos VARCHAR2,p_telefono VARCHAR2)
IS

BEGIN
UPDATE usuario SET correo=LOWER(TRIM(p_correo)),nombre=TRIM(p_nombre),apellidos=TRIM(p_apellidos),telefono=TRIM(p_telefono) WHERE id_usuario=p_id;
IF SQL%ROWCOUNT=0 THEN RAISE_APPLICATION_ERROR(-20001,'Usuario inexistente'); END IF;
END;
/

CREATE OR REPLACE PROCEDURE PR_MT_ESTADO_USUARIO(p_id NUMBER,p_activo NUMBER)
IS

BEGIN
IF p_activo NOT IN(0,1) THEN RAISE_APPLICATION_ERROR(-20002,'Estado inválido'); END IF;
UPDATE usuario SET activo=p_activo WHERE id_usuario=p_id;
END;
/

CREATE OR REPLACE PROCEDURE PR_MT_ASIGNAR_ROL(p_usuario NUMBER,p_rol NUMBER)
IS

BEGIN
MERGE INTO usuario_rol ur USING(SELECT p_usuario id_usuario,p_rol id_rol FROM dual)s
ON(ur.id_usuario=s.id_usuario AND ur.id_rol=s.id_rol)
WHEN MATCHED THEN UPDATE SET ur.activo=1,ur.fecha_asignacion=SYSTIMESTAMP
WHEN NOT MATCHED THEN INSERT(id_usuario,id_rol,activo) VALUES(s.id_usuario,s.id_rol,1);
END;
/
